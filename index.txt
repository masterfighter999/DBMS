    <!-- API-based App Logic -->
    <script type="module">
        // --- Global State ---
        let allBooks = [];
        let allMembers = [];
        let allActiveLoans = [];

        // --- UI Element References ---
        const tabs = document.querySelectorAll('.nav-tab');
        const tabContents = document.querySelectorAll('.tab-content');
        const modal = document.getElementById('modal');
        const modalContent = document.getElementById('modal-content');
        const modalTitle = document.getElementById('modal-title');
        const modalBody = document.getElementById('modal-body');
        const modalCloseBtn = document.getElementById('modal-close-btn');

        // --- Initialization ---
        async function initApp() {
            try {
                // Setup UI listeners
                setupUIListeners();
                
                // Load initial data from API
                await loadAllData();

                // Initialize icons
                if (window.lucide) {
                    window.lucide.createIcons();
                }
            } catch (error) {
                console.error("Error initializing app:", error);
                showToast("Error initializing application.", "error");
            }
        }
        
        async function loadAllData() {
            await Promise.all([
                loadBooks(),
                loadMembers(),
                loadActiveLoans()
            ]);
            updateLoanDropdowns();
        }

        // --- Data Loading Functions ---
        async function loadBooks() {
            const booksTableBody = document.getElementById('books-table-body');
            try {
                const response = await fetch('/api/books');
                if (!response.ok) throw new Error('Failed to fetch books');
                allBooks = await response.json();
                
                renderBooksTable();
            } catch (err) {
                console.error('Error loading books:', err);
                booksTableBody.innerHTML = `<tr><td colspan="6" class="p-4 text-center text-red-500">Error loading books. Is the backend running?</td></tr>`;
            }
        }
        
        async function loadMembers() {
            const membersTableBody = document.getElementById('members-table-body');
            try {
                const response = await fetch('/api/members');
                if (!response.ok) throw new Error('Failed to fetch members');
                allMembers = await response.json();
                
                renderMembersTable();
            } catch (err) {
                 console.error('Error loading members:', err);
                membersTableBody.innerHTML = `<tr><td colspan="5" class="p-4 text-center text-red-500">Error loading members. Is the backend running?</td></tr>`;
            }
        }
        
        async function loadActiveLoans() {
            try {
                const response = await fetch('/api/loans?active=true');
                if (!response.ok) throw new Error('Failed to fetch loans');
                allActiveLoans = await response.json();
                
                renderActiveLoansList();
            } catch (err) {
                console.error('Error loading loans:', err);
                document.getElementById('active-loans-list').innerHTML = `<p class="text-red-500">Error loading loans.</p>`;
            }
        }

        // --- UI Rendering Functions ---
        function renderBooksTable() {
            const booksTableBody = document.getElementById('books-table-body');
            booksTableBody.innerHTML = ''; // Clear loading
            
            if (allBooks.length === 0) {
                booksTableBody.innerHTML = `<tr><td colspan="6" class="p-4 text-center text-gray-500 dark:text-gray-400">No books found. Click 'Add New Book' to start.</td></tr>`;
            } else {
                allBooks.forEach(book => {
                    const tr = document.createElement('tr');
                    tr.className = "hover:bg-gray-50 dark:hover:bg-gray-700";
                    tr.innerHTML = `
                        <td class="px-6 py-4 whitespace-nowrap">
                            <div class="text-sm font-medium text-gray-900 dark:text-white">${book.title}</div>
                        </td>
                        <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500 dark:text-gray-400">${book.author}</td>
                        <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500 dark:text-gray-400">${book.isbn}</td>
                        <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500 dark:text-gray-400">${book.category}</td>
                        <td class="px-6 py-4 whitespace-nowrap">
                            <span class="px-2 inline-flex text-xs leading-5 font-semibold rounded-full ${book.status === 'Available' ? 'bg-green-100 text-green-800 dark:bg-green-900 dark:text-green-200' : 'bg-yellow-100 text-yellow-800 dark:bg-yellow-900 dark:text-yellow-200'}">
                                ${book.status}
                            </span>
                        </td>
                        <td class="px-6 py-4 whitespace-nowrap text-right text-sm font-medium space-x-2">
                            <button class="edit-book-btn text-indigo-600 hover:text-indigo-900 dark:text-indigo-400 dark:hover:text-indigo-200" data-id="${book.id}">Edit</button>
                            <button class="delete-book-btn text-red-600 hover:text-red-900 dark:text-red-400 dark:hover:text-red-200" data-id="${book.id}">Delete</button>
                        </td>
                    `;
                    booksTableBody.appendChild(tr);
                });
            }
        }

        function renderMembersTable() {
            const membersTableBody = document.getElementById('members-table-body');
            membersTableBody.innerHTML = ''; // Clear loading

            if (allMembers.length === 0) {
                membersTableBody.innerHTML = `<tr><td colspan="5" class="p-4 text-center text-gray-500 dark:text-gray-400">No members found. Click 'Add New Member' to start.</td></tr>`;
            } else {
                allMembers.forEach(member => {
                    const joinDate = new Date(member.joinDate).toLocaleDateString();
                    
                    const tr = document.createElement('tr');
                    tr.className = "hover:bg-gray-50 dark:hover:bg-gray-700";
                    tr.innerHTML = `
                        <td class="px-6 py-4 whitespace-nowrap">
                            <div class="text-sm font-medium text-gray-900 dark:text-white">${member.name}</div>
                            <div class="text-sm text-gray-500 dark:text-gray-400">${member.id}</div>
                        </td>
                        <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500 dark:text-gray-400">${member.city}</td>
                        <td class="px-6 py-4 whitespace-nowrap">
                            <span class="px-2 inline-flex text-xs leading-5 font-semibold rounded-full ${member.status === 'A' ? 'bg-green-100 text-green-800 dark:bg-green-900 dark:text-green-200' : 'bg-red-100 text-red-800 dark:bg-red-900 dark:text-red-200'}">
                                ${member.status === 'A' ? 'Active' : 'Suspended'}
                            </span>
                        </td>
                        <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500 dark:text-gray-400">${joinDate}</td>
                        <td class="px-6 py-4 whitespace-nowrap text-right text-sm font-medium space-x-2">
                            <button class="edit-member-btn text-indigo-600 hover:text-indigo-900 dark:text-indigo-400 dark:hover:text-indigo-200" data-id="${member.id}">Edit</button>
                            <button class="delete-member-btn text-red-600 hover:text-red-900 dark:text-red-400 dark:hover:text-red-200" data-id="${member.id}">Delete</button>
                        </td>
                    `;
                    membersTableBody.appendChild(tr);
                });
            }
        }
        
        function renderActiveLoansList() {
            const activeLoansList = document.getElementById('active-loans-list');
            const noActiveLoansMsg = document.getElementById('no-active-loans');
            activeLoansList.innerHTML = '';
            
            if (allActiveLoans.length === 0) {
                activeLoansList.appendChild(noActiveLoansMsg);
                noActiveLoansMsg.style.display = 'block';
                return;
            }
            
            noActiveLoansMsg.style.display = 'none';
            
            allActiveLoans.forEach(loan => {
                // The API response now populates book and member
                const book = loan.book;
                const member = loan.member;
                
                const div = document.createElement('div');
                div.className = "p-4 border rounded-lg flex flex-col sm:flex-row sm:justify-between sm:items-center bg-gray-50 dark:bg-gray-700 dark:border-gray-600";
                div.innerHTML = `
                    <div>
                        <p class="font-semibold text-gray-900 dark:text-white">${book ? book.title : 'Unknown Book'}</p>
                        <p class="text-sm text-gray-600 dark:text-gray-300">Checked out by: <span class="font-medium">${member ? member.name : 'Unknown Member'}</span></p>
                        <p class="text-sm text-gray-500 dark:text-gray-400">Due: ${new Date(loan.dueDate).toLocaleDateString()}</p>
                    </div>
                    <button class="return-book-btn mt-3 sm:mt-0 bg-blue-600 text-white font-bold py-2 px-4 rounded-md hover:bg-blue-700 focus:outline-none focus:ring-2 focus:ring-blue-500 focus:ring-offset-2 dark:focus:ring-offset-gray-800 flex items-center justify-center" data-loan-id="${loan.id}" data-book-id="${book.id}">
                        <i data-lucide="log-in" class="icon mr-2"></i>
                        Return
                    </button>
                `;
                activeLoansList.appendChild(div);
            });
            if (window.lucide) {
                window.lucide.createIcons(); // Re-render icons
            }
        }
        
        // --- UI Event Listeners ---
        function setupUIListeners() {
            // Tab switching
            tabs.forEach(tab => {
                tab.addEventListener('click', () => {
                    tabs.forEach(t => t.classList.replace('active-tab', 'inactive-tab'));
                    tab.classList.replace('inactive-tab', 'active-tab');
                    tabContents.forEach(content => content.classList.add('hidden'));
                    document.getElementById(`${tab.dataset.tab}-tab`).classList.remove('hidden');
                });
            });

            // Theme toggle
            const themeToggle = document.getElementById('theme-toggle');
            const themeIcon = document.getElementById('theme-icon');
            themeToggle.addEventListener('click', () => {
                document.documentElement.classList.toggle('dark');
                if (document.documentElement.classList.contains('dark')) {
                    themeIcon.setAttribute('data-lucide', 'moon');
                    localStorage.setItem('theme', 'dark');
                } else {
                    themeIcon.setAttribute('data-lucide', 'sun');
                    localStorage.setItem('theme', 'light');
                }
                if (window.lucide) window.lucide.createIcons();
            });
            // Apply saved theme on load

                const target = e.target;
                const id = target.dataset.id;
                if (!id) return;
                if (target.classList.contains('edit-member-btn')) {
                    _FORM_ACTIONS.showMemberForm(id);
                } else if (target.classList.contains('delete-member-btn')) {
                    _FORM_ACTIONS.deleteMember(id);

        // --- Loan Dropdown Population ---
        function updateLoanDropdowns() {
            const memberSelect = document.getElementById('checkout-member');
            const bookSelect = document.getElementById('checkout-book');
            
            // Populate members
            memberSelect.innerHTML = '<option value="">Select a member</option>';
            allMembers.forEach(member => {
                if (member.status === 'A') { // Only active members
                    memberSelect.innerHTML += `<option value="${member.id}">${member.name}</option>`;
                }
            });
            if (allMembers.length === 0) memberSelect.innerHTML = '<option value="">No members found</option>';

            // Populate available books
            bookSelect.innerHTML = '<option value="">Select a book</option>';
            const availableBooks = allBooks.filter(book => book.status === 'Available');
            availableBooks.forEach(book => {
                bookSelect.innerHTML += `<option value="${book.id}">${book.title}</option>`;
            });
            if (allBooks.length === 0) bookSelect.innerHTML = '<option value="">No books found</option>';
            else if (availableBooks.length === 0) bookSelect.innerHTML = '<option value="">No books are currently available</option>';
        }

        // --- Action Handlers (Namespaced) ---

        const _FORM_ACTIONS = {
            // --- Book Form ---
            showBookForm: (id = null) => {
                const isEditing = id !== null;
                const book = isEditing ? allBooks.find(b => b.id === id) : {};
                
                modalTitle.textContent = isEditing ? 'Edit Book' : 'Add New Book';
                modalBody.innerHTML = `
                    <form id="book-form" class="space-y-4">
                        <input type="hidden" id="book-id" value="${isEditing ? id : ''}">
                        <div>
                            <label for="book-title" class="form-label">Title</label>
                            <input type="text" id="book-title" class="form-input" required value="${book.title || ''}">
                        </div>
                        <div>
                            <label for="book-author" class="form-label">Author</label>
                            <input type="text" id="book-author" class="form-input" required value="${book.author || ''}">
                        </div>
                        <div>
                            <label for="book-isbn" class="form-label">ISBN</label>
                            <input type="text" id="book-isbn" class="form-input" required value="${book.isbn || ''}">
                        </div>
                        <div>
                            <label for="book-category" class="form-label">Category</label>
                            <input type="text" id="book-category" class="form-input" required value="${book.category || ''}">
                        </div>
                        <button type="submit" class="w-full bg-indigo-600 text-white font-bold py-2 px-4 rounded-md hover:bg-indigo-700 focus:outline-none focus:ring-2 focus:ring-indigo-500 focus:ring-offset-2 dark:focus:ring-offset-gray-800">
                            ${isEditing ? 'Update Book' : 'Add Book'}
                        </button>
                    </form>
                `;
                modal.classList.remove('hidden');
                
                document.getElementById('book-form').addEventListener('submit', _FORM_ACTIONS.handleBookSubmit);
            },
            
            handleBookSubmit: async (e) => {
                e.preventDefault();
                const id = document.getElementById('book-id').value;
                const isEditing = id !== '';
                
                const bookData = {
                    title: document.getElementById('book-title').value,
                    author: document.getElementById('book-author').value,
                    isbn: document.getElementById('book-isbn').value,
                    category: document.getElementById('book-category').value,
                };
                
                try {
                    const url = isEditing ? `/api/books?id=${id}` : '/api/books';
                    const method = isEditing ? 'PUT' : 'POST';
                    
                    if (!isEditing) {
                        bookData.status = 'Available'; // Set default status for new books
                    }
                    
                    const response = await fetch(url, {
                        method: method,
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(bookData)
                    });
                    
                    if (!response.ok) throw new Error(await response.text());
                    
                    showToast(isEditing ? "Book updated!" : "Book added!", "success");
                    modal.classList.add('hidden');
                    loadAllData(); // Refresh all data
                } catch (error) {
                    console.error("Error saving book:", error);
                    showToast("Error saving book.", "error");
                }
            },
            
            deleteBook: async (id) => {
                const book = allBooks.find(b => b.id === id);
                if (book.status === 'On Loan') {
                    showToast("Cannot delete a book that is currently on loan.", "error");
                    return;
                }
                
                if (!await showConfirm("Are you sure you want to delete this book? This action cannot be undone.")) {
                    return;
                }

                try {
                    const response = await fetch(`/api/books?id=${id}`, { method: 'DELETE' });
                    if (!response.ok) throw new Error(await response.text());
                    
                    showToast("Book deleted successfully!", "success");
                    loadAllData(); // Refresh all data
                } catch (error) {
                    console.error("Error deleting book:", error);
                    showToast("Error deleting book.", "error");
                }
            },
            
            // --- Member Form ---
            showMemberForm: (id = null) => {
                const isEditing = id !== null;
                const member = isEditing ? allMembers.find(m => m.id === id) : {};
                
                modalTitle.textContent = isEditing ? 'Edit Member' : 'Add New Member';
                modalBody.innerHTML = `
                    <form id="member-form" class="space-y-4">
                        <input type="hidden" id="member-id" value="${isEditing ? id : ''}">
                        <div>
                            <label for="member-name" class="form-label">Name (UPPERCASE)</label>
                            <input type="text" id="member-name" class="form-input" required value="${member.name || ''}" oninput="this.value = this.value.toUpperCase()">
                        </div>
                        <div>
                            <label for="member-city" class="form-label">City</label>
                            <input type="text" id="member-city" class="form-input" required value="${member.city || ''}">
                        </div>
                        <div>
                            <label for="member-status" class="form-label">Status</label>
                            <select id="member-status" class="form-input">
                                <option value="A" ${member.status === 'A' ? 'selected' : ''}>Active</option>
                                <option value="S" ${member.status === 'S' ? 'selected' : ''}>Suspended</option>
                            </select>
                        </div>
                        <button type="submit" class="w-full bg-indigo-600 text-white font-bold py-2 px-4 rounded-md hover:bg-indigo-700 focus:outline-none focus:ring-2 focus:ring-indigo-500 focus:ring-offset-2 dark:focus:ring-offset-gray-800">
                            ${isEditing ? 'Update Member' : 'Add Member'}
                        </button>
                    </form>
                `;
                modal.classList.remove('hidden');
                
                document.getElementById('member-form').addEventListener('submit', _FORM_ACTIONS.handleMemberSubmit);
            },
            
            handleMemberSubmit: async (e) => {
                e.preventDefault();
                const id = document.getElementById('member-id').value;
                const isEditing = id !== '';
                
                const memberData = {
                    name: document.getElementById('member-name').value,
                    city: document.getElementById('member-city').value,
                    status: document.getElementById('member-status').value,
                };

                if (memberData.name !== memberData.name.toUpperCase()) {
                    showToast("Name must be in UPPERCASE.", "error");
                    return;
                }
                
                try {
                    const url = isEditing ? `/api/members?id=${id}` : '/api/members';
                    const method = isEditing ? 'PUT' : 'POST';

                    if (!isEditing) {
                        memberData.joinDate = new Date().toISOString(); // Set joinDate for new members
                    }

                    const response = await fetch(url, {
                        method: method,
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(memberData)
                    });
                    
                    if (!response.ok) throw new Error(await response.text());
                    
                    showToast(isEditing ? "Member updated!" : "Member added!", "success");
                    modal.classList.add('hidden');
                    loadAllData(); // Refresh all data
                } catch (error) {
                    console.error("Error saving member:", error);
                    showToast("Error saving member.", "error");
                }
            },
            
            deleteMember: async (id) => {
                // Check for active loans before deleting
                // We refetch active loans just to be sure
                let activeLoans = [];
                try {
                    const res = await fetch('/api/loans?active=true');
                    activeLoans = await res.json();
                } catch (e) {
                    showToast("Could not verify active loans. Please try again.", "error");
                    return;
                }

                const hasLoan = activeLoans.some(loan => loan.member.id === id);
                if (hasLoan) {
                    showToast("Cannot delete a member with active loans.", "error");
                    return;
                }

                if (!await showConfirm("Are you sure you want to delete this member?")) {
                    return;
                }

                try {
                    const response = await fetch(`/api/members?id=${id}`, { method: 'DELETE' });
                    if (!response.ok) {
                        const err = await response.json();
                        throw new Error(err.error || 'Failed to delete member');
                    }

                    showToast("Member deleted.", "success");
                    loadAllData(); // Refresh all data
                } catch (error) {
                    console.error("Error deleting member:", error);
                    showToast(error.message || "Error deleting member.", "error");
                }
            },
        };

        const _LOAN_ACTIONS = {
            checkoutBook: async (e) => {
                e.preventDefault();
                const memberId = document.getElementById('checkout-member').value;
                const bookId = document.getElementById('checkout-book').value;
                
                if (!memberId || !bookId) {
                    showToast("Please select both a member and a book.", "error");
                    return;
                }
                
                try {
                    const dueDate = new Date();
                    dueDate.setDate(dueDate.getDate() + 14); // 2-week loan period
                    
                    const response = await fetch('/api/loans', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({
                            memberId: memberId,
                            bookId: bookId,
                            dueDate: dueDate.toISOString(),
                        })
                    });
                    
                    if (!response.ok) throw new Error(await response.text());
                    
                    showToast("Book checked out successfully!", "success");
                    e.target.reset(); // Reset form
                    loadAllData(); // Refresh all data
                } catch (error) {
                    console.error("Error checking out book:", error);
                    showToast(error.message || "Error checking out book.", "error");
                }
            },
            
            returnBook: async (loanId, bookId) => {
                if (!loanId || !bookId) {
                    showToast("Invalid loan or book ID.", "error");
                    return;
                }
                
                try {
                    const response = await fetch(`/api/loans?id=${loanId}`, {
                        method: 'PUT',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ 
                            returnDate: new Date().toISOString(),
                            bookId: bookId // Pass bookId to API to update its status
                        })
                    });
                    
                    if (!response.ok) throw new Error(await response.text());

                    showToast("Book returned successfully!", "success");
                    loadAllData(); // Refresh all data
                } catch (error) {
                    console.error("Error returning book:", error);
                    showToast("Error returning book.", "error");
                }
            }
        };

        function showToast(message, type = "success") {
            const toast = document.getElementById('toast');
            const toastMessage = document.getElementById('toast-message');
            const toastIcon = document.getElementById('toast-icon');
            
            toastMessage.textContent = message;
            
            if (type === 'success') {
                toast.classList.remove('bg-red-600');
                toast.classList.add('bg-gray-900');
                toastIcon.setAttribute('data-lucide', 'check-circle');
            } else {
                toast.classList.remove('bg-gray-900');
                toast.classList.add('bg-red-600');
                toastIcon.setAttribute('data-lucide', 'alert-triangle');
            }
            if (window.lucide) {
                window.lucide.createIcons();
            }
            
            toast.style.opacity = 1;
            setTimeout(() => {
                toast.style.opacity = 0;
            }, 3000);
        }
        
        async function showConfirm(message) {
            return new Promise((resolve) => {
                modalTitle.textContent = "Please Confirm";
                modalBody.innerHTML = `
                    <p class="text-gray-700 dark:text-gray-300 mb-4">${message}</p>
                    <div class="flex justify-end space-x-3">
                        <button id="confirm-cancel" class="bg-gray-200 dark:bg-gray-600 text-gray-800 dark:text-gray-200 font-bold py-2 px-4 rounded-md hover:bg-gray-300 dark:hover:bg-gray-500">Cancel</button>
                        <button id="confirm-ok" class="bg-red-600 text-white font-bold py-2 px-4 rounded-md hover:bg-red-700">Confirm</button>
                    </div>
                `;
                modal.classList.remove('hidden');
                
                const cancelBtn = document.getElementById('confirm-cancel');
                const okBtn = document.getElementById('confirm-ok');
                
                const cleanup = (result) => {
                    modal.classList.add('hidden');
                    resolve(result);
                };
                
                cancelBtn.onclick = () => cleanup(false);
                okBtn.onclick = () => cleanup(true);
            });
        }
